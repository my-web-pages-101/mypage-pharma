            init() {
                // Load saved theme from localStorage
                const savedTheme = localStorage.getItem('pharmacy-theme');
                if (savedTheme && this.themes[savedTheme]) {
                    this.currentTheme = savedTheme;
                }
                this.loadTheme(this.currentTheme);
                this.createThemeSelector();
            },
            
            loadTheme(themeName) {
                const theme = this.themes[themeName];
                if (!theme) return;
                
                this.currentTheme = themeName;
                
                // Save to localStorage
                localStorage.setItem('pharmacy-theme', themeName);
                
                // Load fonts
                const existingLink = document.getElementById('pastel-fonts');
                if (existingLink) existingLink.remove();
                
                const link = document.createElement('link');
                link.id = 'pastel-fonts';
                link.rel = 'stylesheet';
                link.href = theme.fonts.google;
                document.head.appendChild(link);
                
                // Apply CSS variables
                const root = document.documentElement;
                Object.entries(theme.colors).forEach(([key, value]) => {
                    root.style.setProperty(`--color-${key}`, value);
                });
                root.style.setProperty('--theme-radius', theme.borderRadius);
                root.style.setProperty('--theme-shadow', theme.shadow);
                root.style.setProperty('--theme-shadow-hover', theme.shadowHover);
                root.style.setProperty('--font-heading', theme.fonts.heading);
                root.style.setProperty('--font-body', theme.fonts.body);
                
                // Update meta theme color
                document.querySelector('meta[name="theme-color"]').setAttribute('content', theme.colors.background);
                
                // Update body class
                document.body.className = `theme-${themeName}`;
                
                document.dispatchEvent(new CustomEvent('pastelThemeChanged', { detail: { theme: themeName } }));
            },
            
            createThemeSelector() {
                const container = document.createElement('div');
                container.id = 'pastel-theme-selector';
                container.className = 'fixed bottom-6 right-6 z-50';
                
                const btn = document.createElement('button');
                btn.id = 'pastel-toggle';
                btn.className = 'flex items-center gap-3 px-5 py-3.5 rounded-full shadow-lg hover:scale-105 transition-transform duration-200 cursor-pointer';
                btn.setAttribute('aria-label', 'Επιλογή θέματος χρώματος');
                btn.setAttribute('aria-expanded', 'false');
                btn.innerHTML = `
                    <span class="w-6 h-6 rounded-full" style="background: ${this.themes[this.currentTheme].colors.primary}"></span>
                    <span class="hidden sm:inline font-medium text-sm">${this.themes[this.currentTheme].name}</span>
                    <i class="fas fa-chevron-up text-xs transition-transform duration-200" id="pastel-arrow"></i>
                `;
                
                const dropdown = document.createElement('div');
                dropdown.id = 'pastel-dropdown';
                dropdown.className = 'absolute bottom-full right-0 mb-3 rounded-2xl overflow-hidden shadow-2xl opacity-0 invisible transition-all duration-300';
                dropdown.setAttribute('role', 'menu');
                dropdown.setAttribute('aria-label', 'Επιλογή θέματος');
                
                const themeButtons = Object.entries(this.themes).map(([key, theme]) => `
                    <button class="pastel-option w-full px-5 py-3.5 flex items-center gap-3 transition-colors duration-150" data-theme="${key}" role="menuitem">
                        <span class="w-5 h-5 rounded-full flex-shrink-0" style="background: ${theme.colors.primary}"></span>
                        <div class="text-left">
                            <span class="font-medium text-sm block">${theme.name}</span>
                            <span class="text-xs opacity-70">${theme.description}</span>
                        </div>
                    </button>
                `).join('');
                
                dropdown.innerHTML = themeButtons;
                container.appendChild(btn);
                container.appendChild(dropdown);
                document.body.appendChild(container);
                
                const toggle = document.getElementById('pastel-toggle');
                const arrow = document.getElementById('pastel-arrow');
                
                toggle.addEventListener('click', (e) => {
                    e.stopPropagation();
                    const isOpen = !dropdown.classList.contains('opacity-0');
                    dropdown.classList.toggle('opacity-0');
                    dropdown.classList.toggle('invisible');
                    dropdown.setAttribute('aria-expanded', isOpen ? 'false' : 'true');
                    arrow.style.transform = isOpen ? 'rotate(0deg)' : 'rotate(180deg)';
                    if (!isOpen) {
                        dropdown.querySelector('.pastel-option').focus();
                    }
                });
                
                document.addEventListener('click', () => {
                    dropdown.classList.add('opacity-0', 'invisible');
                    dropdown.setAttribute('aria-expanded', 'false');
                    arrow.style.transform = 'rotate(0deg)';
                });
                
                dropdown.querySelectorAll('.pastel-option').forEach(opt => {
                    opt.addEventListener('click', (e) => {
                        e.stopPropagation();
                        this.loadTheme(opt.dataset.theme);
                        dropdown.classList.add('opacity-0', 'invisible');
                        dropdown.setAttribute('aria-expanded', 'false');
                        arrow.style.transform = 'rotate(0deg)';
                        const newTheme = this.themes[opt.dataset.theme];
                        toggle.querySelector('span:first-child').style.background = newTheme.colors.primary;
                        toggle.querySelector('span:nth-child(2)').textContent = newTheme.name;
                        toggle.focus();
                    });
                    
                    // Keyboard navigation
                    opt.addEventListener('keydown', (e) => {
                        if (e.key === 'Enter' || e.key === ' ') {
                            e.preventDefault();
                            opt.click();
                        }
                    });
                });
            }
        };

    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // Initialize theme with localStorage persistence
            PastelThemeManager.init();
            
            // Mobile menu with focus trap
            const mobileBtn = document.getElementById('mobile-menu-btn');
            const mobileMenu = document.getElementById('mobile-menu');
            const mobileLinks = mobileMenu ? mobileMenu.querySelectorAll('a') : [];
            const firstLink = mobileLinks[0];
            const lastLink = mobileLinks[mobileLinks.length - 1];
            
            if (mobileBtn && mobileMenu) {
                mobileBtn.addEventListener('click', () => {
                    const isOpen = mobileMenu.classList.toggle('active');
                    mobileBtn.setAttribute('aria-expanded', isOpen);
                    if (isOpen) {
                        firstLink.focus();
                    }
                });
                
                // Focus trap
                mobileMenu.addEventListener('keydown', (e) => {
                    if (e.key === 'Tab') {
                        if (e.shiftKey && document.activeElement === firstLink) {
                            e.preventDefault();
                            lastLink.focus();
                        } else if (!e.shiftKey && document.activeElement === lastLink) {
                            e.preventDefault();
                            firstLink.focus();
                        }
                    }
                    if (e.key === 'Escape') {
                        mobileMenu.classList.remove('active');
                        mobileBtn.setAttribute('aria-expanded', 'false');
                        mobileBtn.focus();
                    }
                });
                
                mobileLinks.forEach(link => {
                    link.addEventListener('click', () => {
                        mobileMenu.classList.remove('active');
                        mobileBtn.setAttribute('aria-expanded', 'false');
                    });
                });
            }
            
            // Smooth scroll with proper offset
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    e.preventDefault();
                    const targetId = this.getAttribute('href');
                    const target = document.querySelector(targetId);
                    if (target) {
                        const offset = 80;
                        const top = target.getBoundingClientRect().top + window.pageYOffset - offset;
                        window.scrollTo({ top, behavior: 'smooth' });
                    }
                });
            });
            
            // Scroll animations
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('visible');
                    }
                });
            }, { threshold: 0.1, rootMargin: '0px 0px -50px 0px' });
            
            document.querySelectorAll('.fade-up').forEach(el => observer.observe(el));
            
            // Header shadow on scroll
            const header = document.getElementById('header');
            window.addEventListener('scroll', () => {
                if (window.scrollY > 50) {
                    header.style.boxShadow = 'var(--theme-shadow)';
                } else {
                    header.style.boxShadow = 'none';
                }
            }, { passive: true });
            
            // Re-initialize theme
            PastelThemeManager.loadTheme(PastelThemeManager.currentTheme);
        });
    </script>
